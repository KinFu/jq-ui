首先引入jquery库，然后引入tab.js、tab.css，如下： 

<script type="text/javascript" src="js/jquery/jquery-1.7.2.min.js"></script>

<script type="text/javascript" src="js/tab.js"></script>
<link rel="stylesheet" href="css/blue/tab.css"/>

在页面加载完成后初始化tab插件 

<script>
	$(function() {
		$("#tab1").tab();
	});
</script>

页面代码 

<div id="tab1">
	<ul></ul>
	<div></div>
</div>

tab插件有以下函数

函数名 参数 功能 返回值 示例
addTab Object 添加Tab选项卡，{"title": "", "id": "", "content": ""}。
title：标签显示文字；id：页面元素唯一id；content：面板内容 
$('.tab').tab(
	'addTab', 
	{
		'title': '管理员添加', 
		'id': 'tab3', 
		'content': '<p>管理员添加Content</p>'
	}
);
addRemoteTab Object 添加tab选项卡和面板，再从远程url获取数据填充到面板中。
{"title": "", "id": "", "url": "", "method": ""}
title：标签显示文字；id：页面元素唯一id；url：加载数据的地址；method：get/post，默认get 
$('.tab').tab(
	'addRemoteTab', 
	{
		'title': '管理员查看', 
		'id': 'tab2', 
		'url': '/jq-ui/ajax/admin_list.html',
		'method': 'get'
	}
);
removeTab tabId 删除指定选项卡 
$('.tab').tab(
	'removeTab', 'tab1'
);
selectTab tabId 选中指定选项卡 
$('.tab').tab(
	'selectTab', 'tab2'
);
isSelected tabId 判断指定选项卡是否选中 true/false 
$('.tab').tab(
	'isSelected', 'tab2'
);
isExists tabId 判断指定选项卡是否存在 true/false 
$('.tab').tab(
	'isExists', 'tab2'
);
