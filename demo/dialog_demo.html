function openUrlDialog() {
	$.dialog.open({
		id: 'dialog1',
		url: 'ajax/admin_dialog.html',
		width: 600,
		height: 400,
		title: '使用url加载数据',
		modal: true, // true|false
		showType: 'slide' // slide|fade
	});
}
