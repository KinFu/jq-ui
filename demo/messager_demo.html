		title: '信息',
		content: '请填写执行情况(200字以内)',
		level: 'warning', // success|info|question|warning|error
		btn: '知道了',
		time: 1000,
		callback: function() {
			// location.reload();
		},
		showType: 'slide' // slide|fade
	});
}




function openConfirm() {
	$.messager.confirm({
		title: '信息',
		content: '您确定删除员工“John”吗？',
		btn: [ 
		       { text: '确定', callback: function() { 
					$.messager.message({
			   			content: '员工数据删除成功。',
						time: 2000
			   		});
		       } }, 
		       { text: '取消', callback: function() {} }
		]
	});
}
